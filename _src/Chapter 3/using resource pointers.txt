<?
class ResourceObject
{
private $resource;
private $dsn;
public function __construct($dsn)
{
$this->dsn = $dsn;
$this->resource = pg_connect($this->dsn);
}
public function __sleep()
{
pg_close($this->resource);
return array_keys( get_object_vars( $this ) );
}
public function __wakeup()
{
$this->resource = pg_connect($this->dsn);
}
}
?>
